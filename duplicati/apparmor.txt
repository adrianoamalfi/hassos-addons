#include <tunables/global>

profile duplicati flags=(attach_disconnected,mediate_deleted) {
  #include <abstractions/base>

  # Capabilities
  file,
  signal (send) set=(kill,term,int,hup,cont),
  capability dac_read_search,

  # Network access (required for cloud backup targets)
  network inet stream,
  network inet dgram,
  network inet6 stream,
  network inet6 dgram,

  # S6-Overlay
  /init ix,
  /bin/** ix,
  /usr/bin/** ix,
  /run/{s6,s6-rc*,service}/** ix,
  /package/** ix,
  /command/** ix,
  /etc/services.d/** rwix,
  /etc/cont-init.d/** rwix,
  /etc/cont-finish.d/** rwix,
  /run/{,**} rwk,
  /dev/tty rw,

  # Bashio
  /usr/lib/bashio/** ix,
  /tmp/** rwk,

  # Access to options.json and other files within the addon
  /data/** rw,

  # Mono runtime
  /usr/bin/mono ix,
  /usr/lib/mono/** mr,

  # Duplicati
  /usr/lib/duplicati/** mr,
  /usr/bin/duplicati-server ix,
  /usr/bin/start_duplicati ix,

  # Mapped volumes
  /config/** rw,
  /ssl/** rw,
  /media/** rw,
  /addons/** rw,
  /share/** rw,
  /backup/** rw,

  # Start Duplicati service profile
  /usr/bin/duplicati-server cx -> duplicati_server,

  profile duplicati_server flags=(attach_disconnected,mediate_deleted) {
    #include <abstractions/base>

    # Receive signals from S6-Overlay
    signal (receive) peer=*_duplicati,

    # Network access (required for cloud backup targets)
    network inet stream,
    network inet dgram,
    network inet6 stream,
    network inet6 dgram,

    # Access to addon data
    /data/** rw,

    # Access to mapped volumes
    /config/** rw,
    /ssl/** rw,
    /media/** rw,
    /addons/** rw,
    /share/** rw,
    /backup/** rw,

    # Mono runtime and Duplicati binaries
    /usr/bin/mono rix,
    /usr/lib/mono/** mr,
    /usr/lib/duplicati/** mr,
    /usr/bin/duplicati-server r,

    # System access
    /bin/bash rix,
    /bin/echo ix,
    /etc/passwd r,
    /dev/tty rw,
  }
}
