#include <tunables/global>

profile duplicati flags=(attach_disconnected,mediate_deleted) {
  #include <abstractions/base>

  # Capabilities
  file,
  signal (send) set=(kill,term,int,hup,cont),
  capability dac_read_search,
  capability setuid,
  capability setgid,

  # Network access (required for cloud backup targets and nginx proxy)
  network inet stream,
  network inet dgram,
  network inet6 stream,
  network inet6 dgram,

  # S6-Overlay
  /init ix,
  /bin/** ix,
  /usr/bin/** ix,
  /usr/sbin/** rix,
  /run/{s6,s6-rc*,service}/** ix,
  /package/** ix,
  /command/** ix,
  /etc/services.d/** rwix,
  /etc/cont-init.d/** rwix,
  /etc/cont-finish.d/** rwix,
  /run/{,**} rwk,
  /dev/tty rw,

  # Bashio
  /usr/lib/bashio/** ix,
  /tmp/** rwk,

  # Access to options.json, secrets, and other addon files
  /data/** rw,

  # Duplicati (.NET native, self-contained)
  /usr/lib/duplicati/** rix,
  /usr/bin/duplicati-server rix,
  /usr/bin/start_duplicati ix,

  # nginx reverse proxy
  /usr/sbin/nginx rix,
  /usr/bin/start_nginx ix,
  /usr/lib/nginx/** r,
  /usr/share/nginx/** r,
  /etc/nginx/** r,
  /var/log/nginx/** rw,
  /var/lib/nginx/** rw,

  # Mapped volumes
  /config/** rw,
  /ssl/** rw,
  /media/** rw,
  /addons/** rw,
  /share/** rw,
  /backup/** rw,

  # System access
  /etc/passwd r,
  /proc/sys/net/core/somaxconn r,
}
